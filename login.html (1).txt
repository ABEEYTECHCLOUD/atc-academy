      <form id="loginForm">
        <div class="form-row full">
          <label>Email</label>
          <input id="email" type="email" required />
        </div>
        <div class="form-row full">
          <label>Password</label>
          <input id="password" type="password" required />
        </div>
        <div style="margin-top:12px">
          <button class="btn" type="submit">Login</button>
          <a href="index.html" class="btn-ghost" style="margin-left:10px">Register</a>
        </div>
        <div id="loginMsg" class="note"></div>
      </form>
    </div>
  </div>

  <script type="module">
    import { loginStudent, onAuth } from './script.js';
    const form = document.getElementById('loginForm'), msg = document.getElementById('loginMsg');

    form.addEventListener('submit', async (e)=>{
      e.preventDefault();
      msg.textContent = 'Logging in...';
      try{
        const email = document.getElementById('email').value;
        const password = document.getElementById('password').value;
        await loginStudent(email,password);
        msg.innerHTML = '<span class="success">Login successful. Redirecting...</span>';
        setTimeout(()=> location.href = 'dashboard.html', 900);
      }catch(err){
        msg.innerHTML = '<span class="error">'+(err.message||err)+'</span>';
      }
    });

    // if already logged in, go to dashboard
    onAuth(user=>{
      if(user) location.href = 'dashboard.html';
    });
  </script>
